hooks/useBatchCalls.js

import { isEmpty } from "underscore"

import { useSnackbar } from "contexts/SnackbarContext"
import { endpoints } from "data/api_data"

import useHandleRequest from "./useHandleRequest"

/**
 * Custom hook that gives batch API functionality, such as creating multiple
 * items, deleting all items/boxes/pallets, and updating many items.
 * @category Hooks 
 * @module useBatchCalls
 */
const useBatchCalls = () => {
    const handleRequest = useHandleRequest()
    const { triggerSnackbar } = useSnackbar()

    /**
     * Deletes all given PO Items
     * @param {PurchaseItem[]} purchaseItems
     */
    const deletePurchaseItems = async (purchaseItems) => {
        for (const item of purchaseItems) {
            await handleRequest(
                endpoints.purchaseOrderItems.delete(item.id),
                "DELETE",
                undefined
            )
        }
    }

    /**
     * POSTs all given items to the given purchase order ID
     * @param {PurchaseItem[]} purchaseItems 
     * @param {number} purchaseOrderID 
     */
    const postPurchaseItems = async (purchaseItems, purchaseOrderID) => {
        for (const item of purchaseItems) {
            await handleRequest(
                endpoints.purchaseOrderItems.post,
                "POST",
                {
                    ...item,
                    po_id: purchaseOrderID
                }
            )
        }
    }

    /**
     * PUTs all given items to edit them
     * @param {PurchaseItem[]} purchaseItems 
     */
    const putPurchaseItems = async (purchaseItems) => {
        for (const item of purchaseItems) {
            await handleRequest(
                endpoints.purchaseOrderItems.put(item.id),
                "PUT",
                {
                    ...item,
                    description: `${item.description} - ${item.quantity} units`,
                }
            )
        }
    }

    /**
     * Gets the box or pallet data for the given container number
     * 
     * @param {number} containerNumber 
     */
    const getBoxOrPallet = async (containerNumber) => {
        const box = await handleRequest(
            endpoints.boxes.getByID(containerNumber || 0),
            "GET",
            undefined
        )
        if (!isEmpty(box)) return { ...box, is_pallet: false, container_num: containerNumber }

        const pallet = await handleRequest(
            endpoints.pallets.getByID(containerNumber || 0),
            "GET",
            undefined
        )
        if (!isEmpty(pallet)) return { ...pallet, is_pallet: true, container_num: Number(containerNumber) }

        return {}
    }

    /**
     * Gets all items for the given boxes and pallets
     * 
     * @param {Array.<Box | Pallet>} containers 
     */
    const getItemsInContainers = async (containers) => {
        let allItems = []
        for (const container of containers) {
            const items = await handleRequest(
                endpoints.items[`getBy${container.is_pallet ? "Pallet" : "Box"}ID`](container.container_num),
                "GET",
                undefined
            )
            allItems = allItems.concat(items)
        }
        return allItems
    }

    /**
     * Deletes all given PO Items
     * @param {object[]} invoiceItems
     */
    const deleteInvoiceItems = async (invoiceItems) => {
        for (const item of invoiceItems) {
            await handleRequest(
                endpoints.pickTicketItems.delete(item.id),
                "DELETE",
                undefined
            )
        }
    }

    /**
     * POSTs all given items to the given purchase order ID
     * @param {object[]} invoiceItems 
     * @param {number} invoiceID 
     */
    const postInvoiceItems = async (invoiceItems, invoiceID) => {
        let backtrack = []
        for (const item of invoiceItems) {
            const response = await handleRequest(
                endpoints.pickTicketItems.post,
                "POST",
                {
                    ...item,
                    invoice_id: invoiceID
                }
            )
            console.log(response)
            if (isEmpty(response)) {
                triggerSnackbar("Unable to create pick ticket items")
                break
            }
            backtrack.push(response)
        }
        if (backtrack.length !== invoiceItems.length) {
            await deleteInvoiceItems(backtrack)
            await handleRequest(
                endpoints.invoices.delete(invoiceID),
                "DELETE"
            )
            return false
        }
        return true
    }

    /**
     * PUTs all given items to edit them
     * @param {object[]} invoiceItems 
     */
    const putInvoiceItems = async (invoiceItems) => {
        for (const item of invoiceItems) {
            await handleRequest(
                endpoints.pickTicketItems.put(item.id),
                "PUT",
                item
            )
        }
    }

    /**
     * PUTs all given items to edit them
     * @param {ReceivingItem[]} receivingItems 
     */
    const putReceivingItems = async (receivingItems) => {
        for (const item of receivingItems) {
            await handleRequest(
                endpoints.items.put(item.id),
                "PUT",
                item
            )
        }
    }

    return {
        deletePurchaseItems,
        postPurchaseItems,
        putPurchaseItems,
        getBoxOrPallet,
        getItemsInContainers,
        deleteInvoiceItems,
        postInvoiceItems,
        putInvoiceItems,
        putReceivingItems
    }
}

export default useBatchCalls
