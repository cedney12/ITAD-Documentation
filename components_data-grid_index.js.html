import { themeQuartz } from "ag-grid-community"
// eslint-disable-next-line no-unused-vars
import { ColDef } from 'ag-grid-community'
import { AgGridReact } from "ag-grid-react"
import { useEffect, useState } from "react"

import Footer from "./Footer"

/**
 * Data Grid that replaced CustomDataGrid based on MUI's DataGrid. Unfortunately, as nice as MUI is, its DataGrid
 * does not have native (and free) Virtualization, Pagination, and Sorting. With larger databases, as ITAD will most
 * likely grow, all three of these are important for a dynamic and fast website. Therefore, this new Data Grid
 * supports all three quite well. It has the following features:
 * - Abstracts away some default properties
 * - Height is easily-settable and the grid, plus footer, will take up the whole height
 * - Adds a footer with pagination and searching, currently will always exist, but will add a flag to disable footer later
 * - Width resizes to fit the columns passed in
 * 
 * @param {object} props 
 * @param {object[]} props.rows Rows for the data grid
 * @param {number} totalRows The total number of rows overall, normally gotten from an endpoint
 * @param {ColDef[]} columns AG-Grid columns
 * @param {function} getRowId Function to get a unique identifier for the row, defaults to the row's ID field
 * @param {string} height CSS-formatted height, the grid and footer will fit to this value
 * @param {DataGridControls} controls Grid Controls from useDataGridControls
 * @param {boolean} loading `true` if the `rows` data is loading, `false` otherwise
 * @param {function} handleSelectRow Function to handle row selection, normally from useSelectRow
 * 
 * @component
 * 
 * @category Components
 * @subcategory Data Grid
 * @module DataGrid 
 */
const DataGrid = ({
    rows,
    totalRows,
    columns,
    getRowId = (params) => `${params.data.id}`,
    height = "80vh",
    controls,
    loading,
    handleSelectRow
}) => {

    const { paginationModel, setPaginationModel, setSortModel, searchValue, setSearchValue } = controls

    const [calculatedWidth, setCalculatedWidth] = useState(columns.reduce((acc, col) => acc + col.width, 2))

    useEffect(() => {
        const calculateWidth = () => {
            // Starts at 2 for the 1px border on each side
            return columns.reduce((acc, col) => acc + col.width, 2)
        }

        setCalculatedWidth(calculateWidth())
    }, [columns])

    const myTheme = themeQuartz.withParams({
        borderColor: "gray",
        rowBorder: { style: "solid", width: 1 },
        columnBorder: { style: "solid", width: 1 },
    })

    const handleSortChange = (e) => {
        const { columns } = e
        const { colId, sort } = columns[columns.length - 1]
        if (sort == null) {
            setSortModel({
                sortField: null,
                sortDirection: null,
            })
            return
        }
        setSortModel({
            sortField: colId,
            sortDirection: sort
        })
    }

    return (
        <div style={{ height, width: calculatedWidth }} className="data-grid">
            <div style={{ height: `calc(${height} - 50px)`, width: calculatedWidth }}>
                <AgGridReact
                    rowData={rows}
                    columnDefs={columns}
                    theme={myTheme}
                    getRowId={getRowId}
                    suppressPaginationPanel
                    loading={loading}
                    rowSelection={{
                        mode: "singleRow",
                        enableClickSelection: true,
                        checkboxes: false
                    }}
                    onRowSelected={handleSelectRow}
                    onSortChanged={handleSortChange}
                    animateRows={false}
                />
            </div>
            <Footer
                paginationModel={paginationModel}
                setPaginationModel={setPaginationModel}
                totalPages={Math.ceil(totalRows / paginationModel.pageSize)}
                searchValue={searchValue}
                setSearchValue={setSearchValue}
            />
        </div>
    )
}

export default DataGrid
