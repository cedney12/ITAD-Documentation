import { Checkbox, FormControlLabel, MenuItem, TextField } from '@mui/material'
import React, { useState } from 'react'
import { isEmpty } from 'underscore'

import { SubmitButton } from 'components/buttons/Buttons'
import { AddButton, DeleteButton, EditButton } from 'components/buttons/IconButtons'

import styles from './Form.module.css'
import TrayItemDetail from '../text/TrayItemDetail'
import { BiCheck, BiX } from 'react-icons/bi'

/**
 * Editable UI for CPUs, for use with creating and editing a Tray.
 * 
 * @param {object} props
 * @param {TrayItem | {}} props.data CPU to display, default {}
 * @param {React.SetStateAction<React.Dispatch<TrayItem[]>>} props.setItems Function to add/edit the state that holds array of `TrayItem`s
 * 
 * @returns {JSX.Element}
 * @component
 * @category Components
 * @subcategory Form
 * @module CPURow
 */
const CPURow = ({
    data = {},
    setItems
}) => {

    const [editing, setEditing] = useState(false)

    const handleSubmit = (e) => {
        e.preventDefault()
        e.stopPropagation()

        const formData = {
            quantity: e.target.quantity.value,
            type: e.target.type.value,
            model: e.target.model.value,
            speed: e.target.speed.value,
            has_imperfections: e.target.imperfections.checked,
            has_chipped_edges: e.target.chipped_edges.checked
        }
        setItems(prev => [...prev, formData])
        setEditing(false)
    }

    const removeItem = () => {
        setItems(prev => {
            let newItems = []
            for (const item of prev) {
                if (item !== data) {
                    newItems.push(item)
                }
            }
            return newItems
        })
    }

    return editing ?
        <form className={`${styles.trayItemRow} ${styles.cpu}`} onSubmit={handleSubmit}>
            <DeleteButton onClick={() => setEditing(false)} borderColor="transparent" />
            <TextField
                name="quantity"
                label="Qty"
                placeholder="Qty"
                defaultValue={data.quantity ?? ""}
                sx={{ width: "75px" }}
                type="number"
                required
            />
            <TextField
                name="type"
                label="Type"
                placeholder="Type"
                defaultValue={data.type ?? ""}
                sx={{ width: "100px" }}
                select
                required
            >
                <MenuItem value="i3">i3</MenuItem>
                <MenuItem value="i5">i5</MenuItem>
                <MenuItem value="i5">i7</MenuItem>
            </TextField>
            <TextField
                name="model"
                label="Model"
                placeholder="Model"
                defaultValue={data.model ?? ""}
                sx={{ width: "100px" }}
                required
            />
            <TextField
                name="speed"
                label="Speed"
                placeholder="Speed"
                defaultValue={data.speed ?? ""}
                sx={{ width: "100px" }}
                required
            />
            <FormControlLabel
                control={<Checkbox defaultChecked={data.has_imperfections ?? true} name="imperfections" />}
                label="Imperfections?"
                labelPlacement="top"
            />
            <FormControlLabel
                control={<Checkbox defaultChecked={data.has_chipped_edges ?? true} name="chipped_edges" />}
                label="Chipped Edges?"
                labelPlacement="top"
            />
            <SubmitButton>Done</SubmitButton>
        </form>
        :
        !isEmpty(data)
            ?
            <div className={`${styles.trayItemRow} ${styles.cpu}`}>
                <TrayItemDetail label="Quantity" value={data.quantity} />
                <TrayItemDetail label="Type" value={data.type} />
                <TrayItemDetail label="Model" value={data.model} />
                <TrayItemDetail label="Speed" value={`${data.speed} Ghz`} />
                <TrayItemDetail label="Imperfections" value={data.has_imperfections ? <BiCheck /> : <BiX />} />
                <TrayItemDetail label="Chipped Edges" value={data.has_chipped_edges ? <BiCheck /> : <BiX />} />
                <EditButton onClick={() => setEditing(true)} borderColor="transparent" sx={{ marginLeft: "auto" }} />
                <DeleteButton onClick={removeItem} borderColor="transparent" />
            </div>
            :
            <div className={`${styles.trayItemRow} ${styles.cpu}`} onClick={() => setEditing(prev => !prev)} style={{ cursor: "pointer" }}>
                <AddButton borderColor="transparent" />
                <h1>Add CPU</h1>
            </div>

}

export default CPURow
