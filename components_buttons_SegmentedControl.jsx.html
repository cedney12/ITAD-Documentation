/* eslint-disable react-hooks/exhaustive-deps */
import React, { useEffect, useRef, useState } from 'react'

import { capitalize } from 'utils/formatter'

import styles from './Buttons.module.css'

/**
 * Special type of button to select from a given group of options that are visible at all times.
 * The selected option will have a colored pill as a background, which is animated to move as 
 * different options are selected.
 * 
 * @name SegmentedControl
 * @param {object} props
 * @param {string[]} props.options Segmented Control options, i.e. the labels
 * @param {string} props.state Current selected option (for controlled state)
 * @param {React.Dispatch<React.SetStateAction<string>>} props.setState The function to set the selected option (for controlled state)
 * @param {string} props.name The name for the item (for uncontrolled state)
 * @returns {JSX.Element}
 * @category Components
 * @subcategory Buttons
 * @module SegmentedControl
 * @component
 * @example
 * // in a React Component
 * ...
 * const [option, setOption] = useState("A")
 * return (
 *     <SegmentedControl 
 *         options={["A", "B", "C", "D"]}
 *         state={option}
 *         setState={setOption}
 *     />
 * )
 */
const SegmentedControl = ({
    options,
    state,
    setState,
    name
}) => {
    const containerRef = useRef(null)
    const [position, setPosition] = useState(4)
    const [width, setWidth] = useState(60)
    const [localValue, setLocalValue] = useState(state ?? options[0])

    const removeSpaces = (s) => s.replace(" ", "_")

    const handleClick = (newOption) => {
        if (newOption != null) {
            setLocalValue(newOption)
            setState && setState(newOption)
        }
    }

    useEffect(() => {
        if (containerRef.current) {
            const selectedButton = containerRef.current.querySelector(`#${removeSpaces(localValue)}`)
            if (selectedButton) {
                setPosition(selectedButton.offsetLeft)
                setWidth(selectedButton.offsetWidth)
            }
        }
    }, [localValue])

    useEffect(() => {
        setLocalValue(state ?? options[0])
    }, [options])

    return (
        <div className={styles.container} ref={containerRef}>
            <input type="hidden" name={name} value={localValue} />
            {
                options.map((option) => {
                    return (
                        <div
                            className={styles.option}
                            style={{
                                color: localValue === option ? "white" : "black"
                            }}
                            id={removeSpaces(option)}
                            onClick={() => handleClick(option)}
                            key={option}
                        >
                            {capitalize(option)}
                        </div>
                    )
                })
            }
            <div
                className={styles.movingBox}
                style={{
                    left: `${width === 60 ? position : position - 4}px`,
                    width: width === 60 ? width : width + 8
                }}
            >
            </div>
        </div>
    )
}

export default SegmentedControl
