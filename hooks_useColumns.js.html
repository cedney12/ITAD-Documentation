// eslint-disable-next-line no-unused-vars
import { ColDef } from "ag-grid-community"

import OptionalCell from "components/data-grid/OptionalCell"
import NullCell from "components/table/NullCell"
import { formatDate, formatNumber } from "utils/formatter"

import useLocations from "./useLocations"

/**
 * Custom hook that contains all columns for the project. This is for use with
 * CustomDataGrid (an extension of MUI's DataGrid), and custom tables for OrderInfo
 * @returns {Column[]}
 * @category Hooks
 * @module useColumns
 */
const useColumns = () => {
    const { locations } = useLocations()

    /** @type {Object.<string, ColDef>} */
    const columns = {
        first_name: {
            field: "first_name",
            width: 200,
            headerName: "First Name",
        },
        last_name: {
            field: "last_name",
            width: 200,
            headerName: "Last Name",
        },
        email: {
            field: "email",
            width: 200,
            headerName: "Email",
        },
        role: {
            field: "role",
            width: 200,
            headerName: "Role",
        },
        asset_num: (onClick) => ({
            field: "asset_num",
            width: 100,
            headerName: "Asset #",
            cellRenderer: OptionalCell,
            cellRendererParams: (params) => ({
                params,
                onClick,
                clickable: true
            })
        }),
        asset_num_no_select: {
            field: "asset_num",
            width: 100,
            headerName: "Asset #",
        },
        description: {
            field: "description",
            width: 200,
            headerName: "Description"
        },
        condition: {
            field: "condition",
            width: 100,
            headerName: "Condition",
            renderCell: (params) => params.value ?? <NullCell />
        },
        supplier: {
            field: "supplier",
            width: 200,
            headerName: "Supplier",
        },
        date_received: {
            field: "date_received",
            width: 100,
            headerName: "Received",
            valueFormatter: (params) => formatDate(params.value)
        },
        date_shipped: {
            field: "date_shipped",
            width: 100,
            headerName: "Shipped",
            valueFormatter: (params) => formatDate(params.value),
        },
        date_created: {
            field: "date_created",
            width: 100,
            headerName: "Created",
            valueFormatter: (params) => formatDate(params.value),
        },
        box_id: (onClick) => ({
            field: "box_id",
            width: 125,
            headerName: "Box #",
            cellRenderer: OptionalCell,
            cellRendererParams: (params) => ({
                params,
                onClick,
                clickable: true
            })
        }),
        box_num_as_id: (onClick) => ({
            field: "id",
            width: 125,
            headerName: "Box #",
            cellRenderer: OptionalCell,
            cellRendererParams: (params) => ({
                params,
                onClick,
                clickable: true
            })
        }),
        location_id: {
            field: "location_id",
            width: 125,
            headerName: "Location",
            valueFormatter: (params) => locations.find((l) => l.id === Number(params.value))?.location_name ?? "N/A"
        },
        item_count: {
            field: "item_count",
            width: 125,
            headerName: "Item Count"
        },
        pallet_id: (onClick) => ({
            field: "pallet_id",
            width: 125,
            headerName: "Pallet #",
            cellRenderer: OptionalCell,

                params,

            valueFormatter: (params) => <p>{params.value ? "Pallet" : "Box"}</p>,
        },
        needed: {
            field: "needed",
            width: 100,
            headerName: "Needed",

        remaining: {

            cellRendererParams: (params) => ({
                params

        },
        model: {
            field: "model",
            width: 100,
            headerName: "Model"
        },
        type: {
            field: "type",
            width: 100,
            headerName: "Type"
        },
        speed: {
            field: "speed",
            width: 100,
            headerName: "Speed"
        },
        quantity: {
            field: "quantity",
            width: 100,
            headerName: "Quantity"
        },
        tray_num_as_id: (onClick) => ({
            field: "id",
            width: 100,
            headerName: "Tray #",
            cellClass: "clickable",
            cellRenderer: OptionalCell,
            cellRendererParams: (params) => ({
                params,
                onClick,
                clickable: true
            })
        }),
        tray_type: {
            width: 100,
            headerName: "Tray Type",
            valueFormatter: (params) => params.data.has_ram ? "RAM" : "CPU"
