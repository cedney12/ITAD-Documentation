import styles from './Orders.module.css'
import OrderInfoTitle from './OrderInfoTitle'
import PartnerDetails from 'components/text/PartnerDetails'

/**
 * Component to show the details of an order.
 * NOTE: Will not handle undefined order
 * 
 * @param {object} props 
 * @param {"Receiving" | "Purchase" | "Invoice"} props.type Type of order
 * @param {object} props.order Order data
 * @param {object[]} props.items Items for the order
 * @param {boolean} props.loadingItems true if loading the items, false otherwise
 * @param {object} props.titleInfo Data to display in the title 
 * @param {string} props.titleInfo.numberLabel
 * @param {number} props.titleInfo.number
 * @param {string} props.titleInfo.dateLabel
 * @param {string} props.titleInfo.date
 * @param {boolean} [props.titleInfo.isInterco] 
 * @param {object[]} props.columns Custom columns for the items table
 * @returns {JSX.Element}
 * @component
 * @category Components
 * @subcategory Orders
 * @module OrderInfo
 * @example
 * return (
 *     <OrderInfo 
 *         type="Receiving"
 *         order={{...}} // order data
 *         items={[...]} // items data
 *         loadingItems={itemsLoading} // boolean to show if items are loading
 *         titleInfo={{
 *             numberLabel="Order",
 *             number=52101,
 *             dateLabel="Received",
 *             numberLabel=new Date(),
 *         }}
 *     />
 * )
 */
const OrderInfo = ({
    type,
    order,
    titleInfo,
    itemsTable,
}) => {

    console.log(order)

    return (
        <div className={styles.info}>
            <OrderInfoTitle
                {...titleInfo}
            />

            <div className={styles.partners}>
                <PartnerDetails
                    header="Supplier"
                    lines={[
                        `${order.supp_name || "Digital ITAD"} (#${order.supp_num})`,
                        order.supp_street || "10 Fox Industrial Dr",
                        `${order.supp_city || "Madison"}, ${order.supp_state || "IL"} ${order.supp_zip || "626060"}`,
                    ]}
                />
                {
                    type !== "Invoice" &&
                    <PartnerDetails
                        header="Supplier Contact"
                        lines={[
                            `${order.contact_first} ${order.contact_last}`,
                            order.contact_num,
                            order.contact_email
                        ]}
                        disabledText={titleInfo.isInterco && "Supplier Contact not available on Interco Orders"}
                    />
                }
                {
                    (type === "Purchase" || type === "Invoice") &&
                    <PartnerDetails
                        header="Customer"
                        lines={[
                            order.cust_name,
                            order.cust_street,
                            `${order.cust_city}, ${order.cust_state} ${order.cust_zip}`
                        ]}
                    />
                }
                {
                    type === "Invoice" &&
                    <PartnerDetails
                        header="Customer Contact"
                        lines={[
                            `${order.contact_first} ${order.contact_last}`,
                            order.contact_num,
                            order.contact_email
                        ]}
                    />
                }
            </div>
            {itemsTable}
        </div>
    )
}

export default OrderInfo
