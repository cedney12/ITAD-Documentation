components/form/RAMRow.jsx

import { MenuItem, TextField } from '@mui/material'
import { useState } from 'react'
import { BiSolidCheckShield, BiSolidShieldX } from 'react-icons/bi'
import { isEmpty } from 'underscore'

import { SubmitButton } from 'components/buttons/Buttons'
import { AddButton, DeleteButton, EditButton } from 'components/buttons/IconButtons'
import ShieldCheckbox from 'components/form/ShieldCheckbox'
import TrayItemDetail from 'components/text/TrayItemDetail'

import styles from './Form.module.css'

/**
 * Editable UI for RAM sticks, for use with creating and editing a Tray.
 * 
 * @param {object} props
 * @param {TrayItem | {}} props.data CPU to display, default {}
 * @param {React.SetStateAction<React.Dispatch<TrayItem[]>>} props.setItems Function to add/edit the state that holds array of `TrayItem`s
 * 
 * @returns {JSX.Element}
 * @component
 * @category Components
 * @subcategory Form
 * @module RAMRow
 */
const RAMRow = ({
    data = {},
    setItems
}) => {

    const [editing, setEditing] = useState(false)

    const handleSubmit = (e) => {
        e.preventDefault()
        e.stopPropagation()

        const formData = {
            quantity: Number(e.target.quantity.value),
            brand: e.target.brand.value,
            model: e.target.model.value,
            size: Number(e.target.size.value),
            type: e.target.type.value,
            version: e.target.version.value,
            speed: Number(e.target.speed.value),
            form_factor: e.target.form_factor.value,
            is_shielded: e.target.is_shielded.value === "true"
        }
        setItems(prev => {
            if (isEmpty(data)) {
                return [...prev, formData]
            }
            let newItems = [...prev]
            let index = prev.findIndex(i => i === data)
            newItems[index] = formData
            return newItems
        })
        setEditing(false)
    }

    const removeItem = () => {
        setItems(prev => prev.filter(i => i !== data))
    }

    return editing ?
        <form className={`${styles.trayItemRow} ${styles.ram}`} onSubmit={handleSubmit}>
            <DeleteButton onClick={() => setEditing(false)} borderColor="transparent" />
            <TextField
                name="quantity"
                label="Qty"
                placeholder="Qty"
                defaultValue={data.quantity ?? ""}
                sx={{ width: "75px" }}
                type="number"
                slotProps={{
                    htmlInput: {
                        min: 1,
                        max: 50
                    }
                }}
                required
            />
            <TextField
                name="brand"
                label="Brand"
                placeholder="Brand"
                defaultValue={data.brand ?? ""}
                sx={{ width: "100px" }}
                required
            />
            <TextField
                name="model"
                label="Model"
                placeholder="Model"
                defaultValue={data.model ?? ""}
                sx={{ width: "100px" }}
                required
            />
            <TextField
                name="size"
                label="Size"
                placeholder="Size"
                defaultValue={data.size ?? ""}
                sx={{ width: "100px" }}
                select
                required
            >
                <MenuItem value={4}>4GB</MenuItem>
                <MenuItem value={8}>8GB</MenuItem>
                <MenuItem value={16}>16GB</MenuItem>
                <MenuItem value={32}>32GB</MenuItem>
                <MenuItem value={64}>64GB</MenuItem>
            </TextField>
            <TextField
                name="speed"
                label="Speed"
                placeholder="Speed"
                defaultValue={data.speed ?? ""}
                sx={{ width: "100px" }}
                type="number"
                slotProps={{
                    htmlInput: {
                        min: 0
                    }
                }}
                required
            />
            <TextField
                name="type"
                label="Type"
                placeholder="Type"
                defaultValue={data.type ?? ""}
                sx={{ width: "100px" }}
                select
                required
            >
                <MenuItem value="Parity">Parity</MenuItem>
                <MenuItem value="ECC">ECC</MenuItem>
            </TextField>
            <TextField
                name="version"
                label="Version"
                placeholder="Version"
                defaultValue={data.version ?? ""}
                sx={{ width: "100px" }}
                select
                required
            >
                <MenuItem value="DDR3">DDR3</MenuItem>
                <MenuItem value="DDR4">DDR4</MenuItem>
            </TextField>
            <TextField
                name="form_factor"
                label="FF"
                placeholder="FF"
                defaultValue={data.form_factor ?? ""}
                sx={{ width: "120px" }}
                select
                required
            >
                <MenuItem value="DIMM">DIMM</MenuItem>
                <MenuItem value="SODIMM">SODIMM</MenuItem>
            </TextField>
            <ShieldCheckbox
                name="is_shielded"
                defaultChecked={data.is_shielded ?? true}
            />
            <SubmitButton>Done</SubmitButton>
        </form>
        :
        !isEmpty(data)
            ?
            <div className={`${styles.trayItemRow} ${styles.ram}`}>
                <TrayItemDetail label="Quantity" value={data.quantity} />
                <TrayItemDetail label="Brand" value={data.brand} />
                <TrayItemDetail label="Model" value={data.model} />
                <TrayItemDetail label="Size" value={`${data.size}GB`} />
                <TrayItemDetail label="Speed" value={`${data.speed}MHz`} />
                <TrayItemDetail label="Type" value={data.type} />
                <TrayItemDetail label="Version" value={data.version} />
                <TrayItemDetail label="Form Factor" value={data.form_factor} />
                <TrayItemDetail label="Shielded" value={data.is_shielded
                    ?
                    <BiSolidCheckShield style={{ marginLeft: "auto", marginRight: "4px", fontSize: "20px" }} />
                    :
                    <BiSolidShieldX style={{ marginLeft: "auto", marginRight: "4px", fontSize: "20px" }} />} />
                <EditButton onClick={() => setEditing(true)} borderColor="transparent" sx={{ marginLeft: "auto" }} />
                <DeleteButton onClick={removeItem} borderColor="transparent" />
            </div>
            :
            <div className={`${styles.trayItemRow} ${styles.ram}`} onClick={() => setEditing(prev => !prev)} style={{ cursor: "pointer" }}>
                <AddButton borderColor="transparent" />
                <h1>Add RAM Sticks</h1>
            </div>

}

export default RAMRow
