components/text/PalletDetails.jsx

/* eslint-disable react-hooks/exhaustive-deps */
import { endpoints } from 'data/api_data'
import styles from './Text.module.css'
import useFetchData from 'hooks/useFetchData'
import { IoChevronForward } from 'react-icons/io5'
import { useEffect } from 'react'
import { useLoading } from 'contexts/LoadingContext'

/**
 * Content of a modal for viewing pallets in inventory.
 * 
 * Grabs all boxes in the pallet with the given pallet number. If the pallet hasn't been viewed before,
 * caches the box's data into the `dataStore` to remove multiple re-calls to the API.
 * 
 * @param {object} props
 * @param {number} props.palletNumber
 * @param {function} props.pushModal Function to add ("push") a modal onto the stack
 * @param {Array.<object>} props.dataStore Cache for viewing inventory modal data
 * @param {React.Dispatch<React.SetStateAction<Array.<object>>>} props.setDataStore Function to modify `dataStore`
 * @returns {JSX.Element}
 * @component
 * @category Components
 * @subcategory Text
 * @module PalletDetails
 */
const PalletDetails = ({
    palletNumber,
    pushModal,
    dataStore,
    setDataStore
}) => {
    const { apiData, loading, error } = useFetchData(endpoints.boxes.getByPalletID(palletNumber), { sortField: "id" })
    const { startLoading, stopLoading } = useLoading()

    useEffect(() => {
        if (dataStore[palletNumber]) {
            return
        }
        if (loading) startLoading()
        if (!loading) stopLoading()
        if (!loading && !error) setDataStore(prev => ({ ...prev, [palletNumber]: { apiData } }))
    }, [loading])

    return (
        <>
            <h1>Pallet {palletNumber}</h1>
            {
                (!loading && error) &&
                <p style={{ color: "gray" }}>Unable to get boxes in Pallet {palletNumber}</p>
            }
            {
                (!loading && !error) &&
                <div className={styles.assetBoxes}>
                    {
                        apiData.length > 0
                            ?
                            apiData.map((box) => (
                                <div key={box.box_id} className={styles.assetBox} onClick={() => pushModal({ type: "box", num: box.id })}>
                                    <h2>Box {box.id}</h2>
                                    <p>{box.description}</p>
                                    <IoChevronForward size="20px" />
                                </div>
                            ))
                            :
                            <p style={{ color: "gray" }}>No boxes in Pallet {palletNumber}</p>
                    }
                </div>
            }
        </>
    )
}

export default PalletDetails
