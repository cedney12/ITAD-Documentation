const baseURL = process.env.REACT_APP_API_BASE_URL + "/Digital_ITAD"

/**
 * All API endpoints as a JS object.
 * 
 * Essentially a componanion to the SwaggerUI documentation, as
 * the `endpoints` object is a nested object. Each key in `endpoints`
 * is a table, with the value being another object. Each key in this nested object 
 * is the name for an endpoint, with the value either being a string 
 * or a function that returns a string. The functions are for endpoints
 * that are dependent on an ID or number, such as getting an item by its ID or
 * a purchase order by purchase order number.  
 * 
 * @category Data
 * @module API Endpoints
 * @example
 * // With `endpoints` imported...
 * 
 * const { apiData: purchaseOrders } = useFetchData(endpoints.purchaseOrders.getAll) // see `useFetchData`
 * const { apiData: purchaseItems } = useFetchData(endpoints.purchaseOrderItems.getByPOID(5))
 */
export const endpoints = {

    items: {
        delete: (id) => `${baseURL}/items/${id}`,
        getAll: `${baseURL}/items/`,
        getByID: (id) => `${baseURL}/items/item_id/${id}`,
        getByBoxID: (boxID) => `${baseURL}/items/box_id/${boxID}`,
        getByPalletID: (boxID) => `${baseURL}/items/pallet_id/${boxID}`,
        post: `${baseURL}/items/`,
        put: (id) => `${baseURL}/items/${id}`,
        move: (id) => `${baseURL}/items/move/${id}`
    },
    boxes: {
        delete: (id) => `${baseURL}/boxes/${id}`,
        getAll: `${baseURL}/boxes/`,
        getByID: (id) => `${baseURL}/boxes/box_id/${id}`,
        getByPalletID: (palletID) => `${baseURL}/boxes/pallet_id/${palletID}`,
        post: `${baseURL}/boxes/`,
        put: (id) => `${baseURL}/boxes/${id}`,
        putOnPallet: (boxID, palletID) => `${baseURL}/boxes/move_box/${boxID}?pallet_id=${palletID}`,
        takeOffPallet: (boxID) => `${baseURL}/boxes/move_box/${boxID}`
    },
    pallets: {
        delete: (id) => `${baseURL}/pallets/${id}`,
        getAll: `${baseURL}/pallets/`,
        getByID: (id) => `${baseURL}/pallets/${id}`,
        post: `${baseURL}/pallets/`,
        put: (id) => `${baseURL}/pallets/${id}`,
        moveLocation: (palletID, locationID) => `${baseURL}/pallets/move/${palletID}?location_id=${locationID}`
    },
    purchaseOrders: {
        getAll: `${baseURL}/purchase/purchase_orders/`,
        post: `${baseURL}/purchase/purchase_orders/`,
        getByNumber: (number) => `${baseURL}/purchase/purchase_orders/${number}`,
        put: (id) => `${baseURL}/purchase/purchase_orders/${id}`,
        delete: (id) => `${baseURL}/purchase/purchase_orders/${id}`
    },
    purchaseOrderItems: {
        getAll: `${baseURL}/purchase/po_items/`,
        post: `${baseURL}/purchase/po_items/`,
        getByPOID: (id) => `${baseURL}/purchase/po_items/${id}`,
        put: (id) => `${baseURL}/purchase/po_items/${id}`,
        delete: (id) => `${baseURL}/purchase/po_items/${id}`,
    },
    receivingOrders: {
        getAll: `${baseURL}/receive/receiving_orders`,
        post: `${baseURL}/receive/receiving_orders`,
        getByPONumber: (number) => `${baseURL}/receive/po_num/${number}`,
        put: (id) => `${baseURL}/receive/receiving_orders/${id}`,
        delete: (id) => `${baseURL}/receive/receiving_orders/${id}`,
    },
    receivingOrderItems: {
        getAll: `${baseURL}/receive/rec_items/`,
        post: `${baseURL}/receive/rec_items/`,
        getByROID: (id) => `${baseURL}/receive/ro_id/${id}`,
        put: (id) => `${baseURL}/receive/rec_items/${id}`,
        delete: (id) => `${baseURL}/receive/rec_items/${id}`,
    },
    users: {
        getAll: `${baseURL}/user/`,
        getByEmail: (email) => `${baseURL}/user/email/${email}`,
        post: `${baseURL}/user/`,
        put: (id) => `${baseURL}/user/${id}`,
        delete: (id) => `${baseURL}/user/${id}`,
    },
    customers: {
        getAll: `${baseURL}/customer_supplier/?is_supplier=false`,
        getByID: (id) => `${baseURL}/customer_supplier/${id}`,
        post: `${baseURL}/customer_supplier/`,
        put: (id) => `${baseURL}/customer_supplier/${id}`,
        delete: (id) => `${baseURL}/customer_supplier/${id}`
    },
    suppliers: {
        getAll: `${baseURL}/customer_supplier/?is_supplier=true`,
        getByID: (id) => `${baseURL}/customer_supplier/${id}`,
        post: `${baseURL}/customer_supplier/`,
        put: (id) => `${baseURL}/customer_supplier/${id}`,
        delete: (id) => `${baseURL}/customer_supplier/${id}`
    },
    invoices: {
        getAll: `${baseURL}/invoices/`,
        post: `${baseURL}/invoices/`,

        put: (id) => `${baseURL}/invoices/${id}`,
        delete: (id) => `${baseURL}/invoices/${id}`,
    },

        getAll: `${baseURL}/invoices/pt_items/`,

        put: (id) => `${baseURL}/invoices/pt_items/${id}`,
        delete: (id) => `${baseURL}/invoices/pt_items/${id}`,
    }
}
