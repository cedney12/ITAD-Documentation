components/text/TrayDetails.jsx

/* eslint-disable react-hooks/exhaustive-deps */
import { endpoints } from "data/api_data"
import useFetchData from "hooks/useFetchData"

import styles from './Text.module.css'
import { IoChevronForward } from "react-icons/io5"
import { useEffect, useState } from "react"
import { isEmpty } from "underscore"
import { useLoading } from "contexts/LoadingContext"

/**
 * Content of a modal for viewing trays in inventory.
 * 
 * Grabs all tray items in the tray with the given tray number. If the tray hasn't been viewed before,
 * caches the tray's data into the `dataStore` to remove multiple re-calls to the API.
 * 
 * @param {object} props
 * @param {number} props.trayNumber Tray ID, for API use
 * @param {function} props.pushModal Function to add ("push") a modal onto the stack
 * @param {Array.<object>} props.dataStore Cache for viewing inventory modal data
 * @param {React.Dispatch<React.SetStateAction<Array.<object>>>} props.setDataStore Function to modify `dataStore`
 * 
 * @returns {JSX.Element}
 * @component
 * @category Components
 * @subcategory Text
 * @module TrayDetails
 */
const TrayDetails = ({
    trayNumber,
    pushModal,
    dataStore,
    setDataStore
}) => {
    const { apiData, loading, error } = useFetchData(endpoints.trayItems.getPage(0, 25, trayNumber, "id", "asc"))
    const [isRAMTray, setIsRAMTray] = useState(false)

    const { startLoading, stopLoading } = useLoading()

    useEffect(() => {
        if (dataStore[trayNumber]) {
            return
        }
        if (loading) startLoading()
        if (!loading) stopLoading()
        if (!loading && !error) setDataStore(prev => ({ ...prev, [trayNumber]: { apiData } }))
    }, [loading])

    useEffect(() => {
        if (isEmpty(apiData)) return
        setIsRAMTray(apiData[0].is_ram)
    }, [loading])

    return (
        <>
            <h1>Tray {trayNumber}</h1>
            {
                (!loading && error) &&
                <p style={{ color: "gray" }}>Unable to get items in Tray {trayNumber}</p>
            }
            {
                (!loading && !error) &&
                <div className={styles.assetBoxes}>
                    {
                        apiData.length > 0
                            ?
                            apiData.map((item) => {
                                return (
                                    <div key={item.asset_num} className={styles.assetBox} onClick={() => pushModal({ type: "tray_item", item })}>
                                        {
                                            isRAMTray ?
                                                <h2>{item.quantity} {item.size} RAM</h2>
                                                :
                                                <h2>{item.quantity} {item.brand} {item.model} CPU</h2>
                                        }
                                        <p>Asset: {item.asset_num}</p>
                                        <IoChevronForward size="20px" />
                                    </div>
                                )
                            })
                            :
                            <p style={{ color: "gray" }}>No items in Tray {trayNumber}</p>
                    }
                </div>
            }
        </>
    )
}

export default TrayDetails
