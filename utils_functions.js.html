import JsBarcode from "jsbarcode"
import { isArray } from "underscore"

import { object_fields } from "data/api_data"

/**
 * Returns an integer value based on the number of pixels.
 * Currently supports the following units:
 * - px
 * - vh
 * - vw
 * - %
 * - <> (nothing, i.e. just a number)
 * 
 * @param {string} rawHeight Height as a CSS rule
 * @param {React.MutableRefObject<any>} [elementRef] Reference to the element trying to size, used in % cases
 * @returns {number} Height in pixels
 * @category Utility
 * @module Functions
 * @name getHeightInPixels
 */
export const getHeightInPixels = (rawHeight, elementRef) => {
    if (rawHeight.substring(0, 4) === "calc") {
        rawHeight = rawHeight.substring(5, rawHeight.length - 1)
        let parts = rawHeight.split(/[+-]/g).reverse()
        let operations = rawHeight.match(/[+-]/g).reverse()
        let height = getHeightInPixels(parts.pop().trim(), elementRef)
        console.log(height)
        while (parts.length > 0) {
            let nextPart = parts.pop().trim()
            let nextOperation = operations.pop().trim()
            if (nextOperation === "+") {
                height += getHeightInPixels(nextPart, elementRef)
            } else if (nextOperation === "-") {
                height -= getHeightInPixels(nextPart, elementRef)
            }
        }
        return height
    }
    if (rawHeight.endsWith("px")) {
        return Number(rawHeight.slice(0, -2))
    }
    if (rawHeight.endsWith("vh")) {
        const viewportHeight = window.innerHeight
        const proportion = Number(rawHeight.slice(0, -2)) * 0.01
        return Math.ceil(proportion * viewportHeight)
    }
    if (rawHeight.endsWith("vw")) {
        const viewportWidth = window.innerWidth
        const proportion = Number(rawHeight.slice(0, -2)) * 0.01
        return Math.ceil(proportion * viewportWidth)
    }
    if (rawHeight.endsWith("%")) {
        if (!elementRef?.current) return 0
        const parentHeight = elementRef.current.parentElement.offsetHeight
        const proportion = Number(rawHeight.slice(0, -1)) * 0.01
        return parentHeight * proportion
    }
    if (/\d$/.test(rawHeight)) {
        return Number(rawHeight)
    }
}

/**
 * Generates the canvas data for a barcode
 * 
 * Uses the number parameter to generate a barcode with JsBarcode, returning the resulting data url
 * that can be used as an image src. 
 * 
 * @param {number} numberInput number to encode into a barcode
 * @param {boolean} displayValue displays the number underneath the barcode if true, won't display otherwise
 * @returns data url for barcode image
 * @category Utility
 * @module Functions
 * @name getBarcode
 */
export const getBarcode = (numberInput, displayValue = true, options) => {
    let canvas = document.createElement("canvas")
    canvas.width = 144
    canvas.height = 72

    let barWidth = 2
    switch (numberInput.toString().length) {
        case 4:
            barWidth = 3.5
            break
        case 5:
        case 7:
            barWidth = 2.5
            break
        case 6:
            barWidth = 3
            break
        default:
            break
    }

    JsBarcode(canvas, numberInput, {
        height: 72,
        width: barWidth,
        displayValue: displayValue,
        ...options
    })
    const barcode = canvas.toDataURL()
    return barcode
}

/**
 * Goes through the fields of the given object and determines the type
 * of object passed in. Currently supports the following types:
 * 
 * - Item
 * - Box
 * - Pallet
 * 
 * @param {any} obj Object to be identified
 * @returns Type of obj given, based on API shape
 * @category Utility
 * @module Functions
 * @name getAPIObjectType
 */
export const getAPIObjectType = (obj) => {
    if (isArray(obj)) {
        obj = obj[0]
    }
    const objKeys = Object.keys(obj)
    for (const entry of Object.entries(object_fields)) {
        const key = entry[0]
        const value = entry[1]
        if (value.every(v => objKeys.includes(v))) {
            return key
        }
    }
    return undefined
}

/**
 * Parses the Purchase Order Item description.
 * 
 * Purchase Order Items have a description normally in the form of:
 * "<Item Type> - <Units> units". This function parses it and returns the text and number of units
 * in an array to be deconstructed.
 * 
 * @param {string} description 
 * @returns `array` with `array[0]` as the item type and `array[1]` as the number of units
 */
export const parsePurchaseOrderItemDescription = (description) => {
    const split = description.split(" - ")
    const text = split[0]
    const units = Number(split[1].split(" ")[0])
    return [text, units]
}
