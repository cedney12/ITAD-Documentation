/* eslint-disable react-hooks/exhaustive-deps */
import { Document, Image, Page, PDFViewer, Text, View } from '@react-pdf/renderer'
import { useEffect, useState } from 'react'
import { isEmpty } from 'underscore'

import itadLogo from 'assets/Digital-ITAD-Logo.png'
import { useLoading } from 'contexts/LoadingContext'
import { endpoints } from 'data/api_data'
import { reports_styles } from 'data/reports_data'
import useFetchData from 'hooks/useFetchData'
import useHandleRequest from 'hooks/useHandleRequest'

import PlaceholderPDF from './PlaceholderPDF'

/**
 * Letter-size report for an Invoice's Packing List
 * 
 * Includes all pallets and boxes, separated into two tables,
 * displaying pallet/box numbers and their content descriptions.
 * 
 * @param {object} props
 * @param {number} props.invoiceNumber
 * @param {function} props.updateInstance Function to update the PDF instance for printing
 * 
 * @returns {JSX.Element}
 * @component
 * 
 * @category Components
 * @subcategory Reports
 * @module PackingList
 */
const PackingList = ({
    invoiceNumber,
    updateInstance
}) => {

    const styles = reports_styles['Packing List']
    const handleRequest = useHandleRequest()
    const { startLoading, stopLoading } = useLoading()
    const { apiData: invoiceItems, error } = useFetchData(endpoints.pickTicketItems.getByInvoiceID(invoiceNumber))

    const [tickets, setTickets] = useState([])
    const [doc, setDoc] = useState(null)

    // Adds description of box/pallet
    useEffect(() => {
        if (isEmpty(invoiceItems)) return

        startLoading()
        const preprocessData = async (items) => {
            const promises = Promise.all(items.map(async (i) => {
                const container = await handleRequest(
                    endpoints[i.is_pallet ? "pallets" : "boxes"].getByID(i.container_num),
                    "GET"
                )
                if (isEmpty(container)) return {}

                return {
                    ...i,
                    description: container.description
                }
            }))
            return await promises
        }
        preprocessData(invoiceItems)
            .then((data) => {
                setTickets(data)
                stopLoading()
            })

    }, [invoiceItems])

    useEffect(() => {
        const getDoc = () => {
            if (isEmpty(tickets)) return null

            return (
                <Document>
                    <Page style={styles.page} size="LETTER">
                        <View style={styles.header}>
                            <Image src={itadLogo} style={styles.logo} />
                            <Text><Text style={{ fontWeight: "normal" }}>Invoice:</Text> {invoiceNumber}</Text>
                        </View>
                        <View>
                            <Text style={{ fontWeight: "bold", marginBottom: "4px", fontSize: "24px" }}>Packing List</Text>
                            {/* Pallets */}
                            {
                                tickets.filter(t => t.is_pallet).length > 0 &&
                                <>
                                    <View><Text style={styles.tableTitle}>Pallets</Text></View>
                                    {/* Table Header */}
                                    <View style={styles.tableHeader}>
                                        <Text style={{ width: "10%" }}>Pallet #</Text>
                                        <Text style={{ width: "90%", textAlign: "right" }}>Description</Text>
                                    </View>

                                    {/* Table Data */}
                                    <View>
                                        {
                                            tickets.filter(t => t.is_pallet).map((item, ind) => {
                                                return (
                                                    <View
                                                        style={ind % 2 === 1 ?
                                                            [styles.dataRow, styles.alternate]
                                                            :
                                                            [styles.dataRow]}
                                                        key={ind}
                                                        wrap={false}>
                                                        <Text style={{ width: "10%" }}>{item.container_num}</Text>
                                                        <Text style={{ width: "90%", textAlign: "right" }}>{item.description}</Text>
                                                    </View>
                                                )
                                            })
                                        }
                                    </View>
                                </>
                            }

                            {/* Boxes */}
                            {
                                tickets.filter(t => !t.is_pallet).length > 0 &&
                                <>
                                    <View><Text style={styles.tableTitle}>Boxes</Text></View>
                                    {/* Table Header */}
                                    <View style={styles.tableHeader}>
                                        <Text style={{ width: "10%" }}>Box #</Text>
                                        <Text style={{ width: "90%", textAlign: "right" }}>Description</Text>
                                    </View>

                                    {/* Table Data */}
                                    <View>
                                        {
                                            tickets.filter(t => !t.is_pallet).map((item, ind) => {
                                                return (
                                                    <View
                                                        style={ind % 2 === 1 ?
                                                            [styles.dataRow, styles.alternate]
                                                            :
                                                            [styles.dataRow]}
                                                        key={ind}
                                                        wrap={false}>
                                                        <Text style={{ width: "10%" }}>{item.container_num}</Text>
                                                        <Text style={{ width: "90%", textAlign: "right" }}>{item.description}</Text>
                                                    </View>
                                                )
                                            })
                                        }
                                    </View>
                                </>
                            }
                        </View>
                    </Page>
                </Document>
            )
        }

        const newDoc = getDoc()
        setDoc(newDoc)
        updateInstance(newDoc)
    }, [tickets])

    if (doc) {
        return (
            <PDFViewer width="100%" height="100%">
                {doc}
            </PDFViewer>
        )
    }

    return <PlaceholderPDF error={error} />
}

export default PackingList
