/**
 * @category Components
 * @subcategory Buttons
 * @module IconButtons

import { IconButton } from "@mui/material"
import { BiArchiveIn, BiCheck, BiPencil, BiPlus, BiPrinter, BiRefresh, BiTrash, BiX } from 'react-icons/bi'
import { IoClipboard } from "react-icons/io5"

/**
 * Custom Icon Button that gives the button a 2px-wide border of the given color.
 * 
 * @name BorderIconButton
 * @param {object} props
 * @param {function} [props.onClick] onClick action, default `() => {}`.
 * @param {boolean} [props.disabled] `true` if disabled, `false` otherwise.
 * @param {string} [props.borderColor] Color for the circular border, default `"#707070"`.
 * @param {string} [props.type="button"] HTML button type, default `"button"`.
 * @param {JSX.Element} props.children Icon to be used.
 * @returns {JSX.Element}
 * @component
 */
const BorderIconButton = ({
    onClick = () => { },
    disabled = false,
    borderColor = "#707070",
    type = "button",
    sx,
    children
}) => {
    const calculatedBorderColor = disabled && !borderColor ? "#a6a6a6" : borderColor

    return (
        <IconButton
            onClick={onClick}
            disabled={disabled}
            style={{ boxSizing: "border-box", border: `2px solid ${calculatedBorderColor}`, opacity: disabled ? "0.4" : "1", aspectRatio: 1 }}
            sx={sx}
            type={type}
        >
            {children}
        </IconButton>
    )
}
