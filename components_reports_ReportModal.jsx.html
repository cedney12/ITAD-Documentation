/* eslint-disable react-hooks/exhaustive-deps */
import { Modal } from "@mui/material"
import { Document, Font, usePDF } from "@react-pdf/renderer"
import printJS from "print-js"
import { useEffect, useState } from "react"

import interBold from 'assets/Inter-Bold.ttf'
import interReg from 'assets/Inter-Regular.ttf'
import interSemi from 'assets/Inter-SemiBold.ttf'
import { PrintButton, ReportCloseButton } from "components/buttons/IconButtons"
import { useSnackbar } from "contexts/SnackbarContext"

import BoxLabel from "./BoxLabel"
import Invoice from "./Invoice"
import PackingList from "./PackingList"
import PalletLabel from "./PalletLabel"
import PickTicket from "./PickTicket"
import PurchaseOrder from "./PurchaseOrder"
import styles from './Reports.module.css'

/**
 * Modal for displaying a Report in PDF form
 * 
 * Report type is decided by the `reportLabel` field, with the PDF
 * being populated by the corresponding component. Gives the ability
 * to print the given report from this modal. 
 * 
 * @param {object} props
 * @param {boolean} props.open Whether the modal is open
 * @param {function} props.onClose Function to close the modal
 * @param {string} props.reportLabel Label of the type of report to display
 * @param {ReportInput[]} props.inputs Number inputs for the report
 * 
 * @returns {JSX.Element}
 * @component
 * @category Components
 * @subcategory Reports
 * @module ReportLabel
 */
const ReportModal = ({
    open,
    onClose,
    reportLabel,
    inputs
}) => {

    Font.register({
        family: "Inter",
        fonts: [
            {
                src: interReg,
                fontWeight: "normal"
            }, {
                src: interSemi,
                fontWeight: "semibold"
            }, {
                src: interBold,
                fontWeight: "bold"
            }
        ]
    })

    const [docInstance, updateInstance] = usePDF({ document: <Document /> })
    const [pdfDocument, setDocument] = useState(<Document />)

    const findInputValue = (label) => inputs.find(i => i.label === label)?.value ?? 0
    const { triggerSnackbar } = useSnackbar()

    // Chooses document based on report label
    useEffect(() => {
        let selectedDoc = pdfDocument
        switch (reportLabel) {
            case "Box Label":
                selectedDoc = (
                    <BoxLabel
                        boxNumber={findInputValue("Box Number")}
                        updateInstance={updateInstance}
                    />
                )
                break
            case "Pallet Label":
                selectedDoc = (
                    <PalletLabel
                        palletNumber={findInputValue("Pallet Number")}
                        updateInstance={updateInstance}
                    />
                )
                break
            case "Invoice":
                selectedDoc = (
                    <Invoice
                        invoiceNumber={findInputValue("Invoice Number")}
                        updateInstance={updateInstance}
                    />
                )
                break
            case "Pick Ticket":
                selectedDoc = (
                    <PickTicket
                        invoiceNumber={findInputValue("Invoice Number")}
                        updateInstance={updateInstance}
                    />
                )
                break
            case "Packing List":
                selectedDoc = (
                    <PackingList
                        invoiceNumber={findInputValue("Invoice Number")}
                        updateInstance={updateInstance}
                    />
                )
                break
            case "Purchase Order":
                selectedDoc = (
                    <PurchaseOrder
                        orderNumber={findInputValue("Purchase Order Number")}
                        updateInstance={updateInstance}
                    />
                )
                break
            default:
                triggerSnackbar("Invalid Report Type")
                onClose()
        }
        setDocument(selectedDoc)
    }, [reportLabel, inputs])

    const printPDF = () => printJS(URL.createObjectURL(docInstance.blob))

    return (
        <Modal open={open} onClose={onClose}>
            <div className={styles.modal}>
                <header className={styles.modalHeader}>
                    <PrintButton onClick={printPDF} />
                    <h1>{reportLabel}</h1>
                    <ReportCloseButton onClick={onClose} />
                </header>
                <main id={styles.modalReport}>
                    {pdfDocument}
                </main>
            </div>
        </Modal>
    )
}

export default ReportModal
