import { DataGrid } from "@mui/x-data-grid"
import { useEffect, useRef, useState } from "react"

import { getHeightInPixels } from "utils/functions"

/**
 * Custom DataGrid component, based on MUI's DataGrid.
 * Key functionalities:
 * - Abstracts away some default properties
 * - Adds easily-settable height, which is then recalculated 
 *      to the nearest number of rows
 * - Configures slotProps
 * 
 * @param {CustomDataGridProps} props 
 * @param {object[]} props.rows
 * @param {Column[]} props.columns
 * @param {boolean} props.loading True if data is loading, false otherwise
 * @param {function} props.handleSelectItem Function to set state when a row is selected
 * @param {function} props.getRowId Function that sets the rows IDs (needs to
 *       param                           return a unique field)
 * @param {string} props.sortingField Field of a row the grid sorts by default
 * @param {string} props.sortingDirection Direction of the default sorting
 * @param {string} props.height CSS-style height declaration.
 * 
 * NOTE: Only use % if parent has a set height (ex. 1000px, not 100%)
 * @param {boolean} props.hideFooter
 * @param {boolean} props.checkboxSelection
 * @param {boolean} props.disableColumnMenu
 * @param {boolean} props.disableMultipleRowSelection
 * @param {boolean} props.autoPageSize
 * @returns {JSX.Element}
 * @component
 * @category Components
 * @subcategory Table
 * @module CustomDataGrid
 * @example
 * 
 * const { apiData, loading } = useFetchData(<API endpoint>) // see `useFetchData`
 * const { columns } = useColumns() // see `useColumns`
 * const { handleSelectRow } = useSelectRow(apiData) // see `useSelectRow`
 * 
 * return (
 *     <CustomDataGrid 
 *         rows={apiData}
 *         columns={[
 *             columns.id,
 *             ... // add more columns
 *         ]}
 *         loading={loading}
 *         handleSelectItem={handleSelectRow}
 *         height="70vh"
 *         checkboxSelection
 *     />
 * )
 */
const CustomDataGrid = ({
    rows,
    columns,
    handleSelectItem = () => { },
    handleSelectCell = null,
    sortingField = "id",
    sortingDirection = "asc",
    height = "620px",
    checkboxSelection,
    sx,
    ...rest
}) => {
    const divRef = useRef(null)
    const [calculatedHeight, setCalculatedHeight] = useState(height)
    const [calculatedWidth, setCalculatedWidth] = useState("fit-content")

    // Calculates height to the nearest row (floored)
    useEffect(() => {
        const pixelHeight = getHeightInPixels(height, divRef)
        const headerFooterHeight = 95
        const availableHeight = pixelHeight - headerFooterHeight
        const rowsCanFit = Math.floor(availableHeight / 35)
        setCalculatedHeight((rowsCanFit * 35 + 95) + "px")
    }, [height])

    // Calculates dynamic width
    useEffect(() => {
        setCalculatedWidth(columns.reduce((acc, column) => acc + column.width, checkboxSelection ? 52 : 2))
    }, [checkboxSelection, columns])

    return (
        <div
            ref={divRef}
            style={{
                height: calculatedHeight,
                width: calculatedWidth,
                overflow: "hidden",
            }}
        >
            <DataGrid
                rows={rows} //important
                columns={columns} //important
                checkboxSelection={checkboxSelection}
                onCellClick={(params, event) => handleSelectCell ? handleSelectCell(params, event) : handleSelectItem(params.row, event)} //important - Edit By Ali
                onRowClick={(params, event) => handleSelectItem(params.row, event)}
                rowHeight={35}
                columnHeaderHeight={40}
                sortModel={[{
                    field: sortingField,
                    sort: sortingDirection
                }]}
                slotProps={{
                    loadingOverlay: {
                        variant: "linear-progress",
                        noRowsVariant: "linear-progress",
                    },
                }}
                sx={{
                    background: "white",
                    width: calculatedWidth,
                    "& .scrollDiv": { display: "none" }, // Gets rid of scrollbars
                    overflow: "hidden",
                    "& .MuiDataGrid-selectedRowCount": { display: "none" },
                    ...sx
                }}
                showCellVerticalBorder
                showColumnVerticalBorder
                disableMultipleRowSelection
                {...rest}
            />
        </div>
    )
}

export default CustomDataGrid
