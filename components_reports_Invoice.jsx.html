/* eslint-disable react-hooks/exhaustive-deps */
import { Document, Image, Page, PDFViewer, Text, View } from '@react-pdf/renderer'
import { useEffect, useState } from 'react'
import { isEmpty } from 'underscore'

import itadLogo from 'assets/Digital-ITAD-Logo.png'
import { useLoading } from 'contexts/LoadingContext'
import { endpoints } from 'data/api_data'
import { reports_styles } from 'data/reports_data'
import useBatchCalls from 'hooks/useBatchCalls'
import useFetchData from 'hooks/useFetchData'
import useItems from 'hooks/useItems'
import { formatDate, formatPrice } from 'utils/formatter'

import InfoBox from './InfoBox'
import PlaceholderPDF from './PlaceholderPDF'

/**
 * Letter-size report for an Invoice
 * 
 * Includes information regarding all individual assets in an invoice
 * with their asset number, description, and price.
 * Also displays summed price for all assets.
 * 
 * @param {object} props 
 * @param {number} props.invoiceNumber
 * @param {function} props.updateInstance Function to update the PDF instance for printing
 * 
 * @returns {JSX.Element}
 * @component
 * @category Components
 * @subcategory Reports
 * @module Invoice
 */
const Invoice = ({
    invoiceNumber,
    updateInstance
}) => {

    const styles = reports_styles['Pick Ticket']
    const { startLoading, stopLoading } = useLoading()
    const { prices } = useItems()
    const { getItemsInContainers } = useBatchCalls()
    const { apiData: invoiceData } = useFetchData(endpoints.invoices.getByID(invoiceNumber))
    const { apiData: invoiceItems, error } = useFetchData(endpoints.pickTicketItems.getByInvoiceID(invoiceNumber))

    const [assets, setAssets] = useState([])
    const [doc, setDoc] = useState(null)

    // Add prices to assets
    useEffect(() => {
        if (isEmpty(invoiceItems)) return

        startLoading()

        const preprocessData = async (items) => {
            const assets = await getItemsInContainers(items)
            return assets.map(i => ({
                ...i,
                price: prices[i.description]
            }))
        }
        preprocessData(invoiceItems)
            .then((data) => {
                setAssets(data)
            })

        stopLoading()
    }, [invoiceItems])

    useEffect(() => {
        const getDoc = () => {
            if (isEmpty(assets)) return null

            return (
                <Document>
                    <Page style={styles.page} size="LETTER">
                        <View style={styles.header}>
                            <Image src={itadLogo} style={styles.logo} />
                            <Text>Invoice</Text>
                        </View>
                        <View style={styles.info}>
                            <InfoBox
                                label="Supplier:"
                                text={[
                                    invoiceData.supp_name,
                                    invoiceData.supp_street,
                                    invoiceData.supp_city + ", " + invoiceData.supp_state + " " + invoiceData.supp_zip
                                ]}
                                stylex={{ box: { width: "23.33%", height: "64px" } }}
                            />
                            <InfoBox
                                label="Supplier Contact:"
                                text={[
                                    invoiceData.contact_first + " " + invoiceData.contact_last,
                                    invoiceData.contact_email,
                                    invoiceData.contact_num
                                ]}
                                stylex={{ box: { width: "23.33%", height: "64px" } }}
                            />
                            <InfoBox
                                label="Customer:"
                                text={[
                                    invoiceData.cust_name,
                                    invoiceData.cust_street,
                                    invoiceData.cust_city + ", " + invoiceData.cust_state + " " + invoiceData.cust_zip
                                ]}
                                stylex={{ box: { width: "23.33%", height: "64px" } }}
                            />

                            <View style={{ width: "30%", display: "flex", flexDirection: "column", gap: "8px" }}>
                                <View style={{ display: "flex", flexDirection: "row" }}>
                                    <InfoBox label="Invoice Number" text={[`${invoiceNumber}`]} stylex={{ box: { width: "55%" }, text: { textAlign: "center" }, label: { textAlign: "center" } }} />
                                    <InfoBox label="Created" text={[formatDate(invoiceData.date_created)]} stylex={{ box: { width: "45%" }, text: { textAlign: "center" }, label: { textAlign: "center" } }} />
                                </View>
                            </View>
                        </View>
                        <View>
                            {/* Table Header */}
                            <View style={styles.tableHeader}>
                                <Text style={{ width: "10%" }}>Asset #</Text>
                                <Text style={{ width: "65%" }}>Description</Text>
                                <Text style={{ width: "25%", textAlign: "right" }}>Price</Text>
                            </View>

                            {/* Table Data */}
                            <View>
                                {
                                    assets.map((item, ind) => {
                                        return (
                                            <View
                                                style={ind % 2 === 1 ?
                                                    [styles.dataRow, styles.alternate]
                                                    :
                                                    [styles.dataRow]}
                                                key={ind}
                                                wrap={false}
                                            >
                                                <Text style={{ width: "10%" }}>{item.asset_num}</Text>
                                                <Text style={{ width: "65%" }}>{item.description}</Text>
                                                <Text style={{ width: "25%", textAlign: "right" }}>{formatPrice(item.price)}</Text>
                                            </View>
                                        )
                                    })
                                }
                            </View>
                            <View style={{ ...styles.dataRow, fontWeight: "bold", backgroundColor: "#d9d9d9", borderTop: "1px solid black" }}>
                                <Text style={{ width: "10%" }}></Text>
                                <Text style={{ width: "65%" }}></Text>
                                <Text style={{ width: "25%", textAlign: "right" }}>
                                    Total:  {formatPrice(assets.reduce((sum, i) => sum += i.price, 0))}</Text>
                            </View>
                        </View>
                    </Page>
                </Document >
            )
        }

        const newDoc = getDoc()
        setDoc(newDoc)
        updateInstance(newDoc)
    }, [assets])

    if (doc) {
        return (
            <PDFViewer width="100%" height="100%">
                {doc}
            </PDFViewer>
        )
    }

    return <PlaceholderPDF error={error} />
}

export default Invoice
